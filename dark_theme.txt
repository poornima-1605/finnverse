base.html(before body)


<script>
document.addEventListener("DOMContentLoaded", function() {
    const themeToggle = document.getElementById('theme-toggle');
    const body = document.body;

    if(!themeToggle) return;  // safety check

    // Apply saved theme from localStorage
    if(localStorage.getItem('theme') === 'dark'){
        body.classList.add('dark-theme');
        themeToggle.innerText = "Light Mode";
    }

    // Toggle theme on button click
    themeToggle.addEventListener('click', () => {
        body.classList.toggle('dark-theme');

        if(body.classList.contains('dark-theme')){
            localStorage.setItem('theme', 'dark');
            themeToggle.innerText = "Light Mode";
        } else {
            localStorage.setItem('theme', 'light');
            themeToggle.innerText = "Dark Mode";
        }

        // Optional: Update Chart.js charts dynamically
        if(window.myCharts){
            window.myCharts.forEach(chart => {
                chart.options.plugins.legend.labels.color = body.classList.contains('dark-theme') ? '#e0e0e0' : '#000';
                chart.options.scales.x.ticks.color = body.classList.contains('dark-theme') ? '#e0e0e0' : '#000';
                chart.options.scales.y.ticks.color = body.classList.contains('dark-theme') ? '#e0e0e0' : '#000';
                chart.update();
            });
        }
    });
});
</script>
